import { useState } from 'react';

function Counter() {

  const [count, setCount] = useState(0);

  const increment = () => {
    setCount(count + 1);
    console.log('Increment Callback value', count + 1);
  };

  const decrement = () => {
    setCount(count - 1);
    console.log('Decrement Callback value', count - 1);
  };

  return (
    <div>
      <h1>Counter - {count}</h1>
      <button onClick={increment}>Increment</button>
      <button onClick={decrement}>Decrement</button>
    </div>
  );
  
}

export default Counter;
